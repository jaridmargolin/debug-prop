(constant) debugProp

Source:
  • debug-prop.js, line 37
Easily debug property changes on an object. By default debugging is disabled on the object. To enable debugging, the object must set `opts.debug` to tru.
Examples
const obj = debugProp('state', {
  state: 'test',
  opts: { debug: true}
})

obj.state = 'test2'
// logs => state: test2
const obj = debugProp('state', {
  uuid: '1',
  state: 'test',
  opts: { debug: true}
}, function (val) => console.log(`${uuid}-state: ${val}`))

obj.state = 'test2'
// logs => 1-state: test2
